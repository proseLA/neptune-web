Chips

The Chips component can be used for making a choice, or as a filter with multiple options. See usage guidelines for more info.

Choice

The user can only select one chip with the selected prop as a single value.

() => {
  const [selected, setSelected] = React.useState(1);
  return (
    <Chips
      chips={[
        {
          value: 1,
          label: '100 GBP',
        },
        {
          value: 2,
          label: '200 GBP',
        },
        {
          value: 3,
          label: '300 GBP+',
        },
      ]}
      onChange={({ selectedValue }) => setSelected(selectedValue)}
      selected={selected}
      aria-label="Choose category"
    />
  );
};

Filter

The user can select any number of chips with the selected prop as an array.

() => {
  const [selected, setSelected] = React.useState(['accounting']);
  const handleOnChange = ({ selectedValue, isEnabled }) => {
    setSelected(isEnabled ?[...selected, selectedValue] : selected.filter(value => value !== selectedValue));
  }
  return (
    <Chips
      chips={[
        {
          value: 'accounting',
          label: 'Accounting',
        },
        {
          value: 'payroll',
          label: 'Payroll',
        },
        {
          value: 'reporting',
          label: 'Reporting',
        },
        {
          value: 'payments',
          label: 'Payments',
        },
      ]}
      onChange={handleOnChange}
      selected={selected}
      aria-label="Choose category"
      multiple
    />
  );
};
